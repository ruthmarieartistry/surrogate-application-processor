        input.Candidate_Summary = formattedSummary;

            flagsList.add("Preterm History");
        }

        // Set the flags multi-select field
        input.Application_Flags = flagsList;

        // Set gold star checkbox
        input.Is_Gold_Star_Applicant = isGoldStar;
